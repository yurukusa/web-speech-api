// Speech Recognition — basic usage
const recognition = new (window.SpeechRecognition
  || window.webkitSpeechRecognition)();

recognition.lang = 'en-US';
recognition.interimResults = true;
recognition.continuous = false;

recognition.onresult = (event) => {
  for (const result of event.results) {
    const text = result[0].transcript;
    const conf = result[0].confidence;
    console.log(`${result.isFinal ? '✅' : '⏳'} ${text} (${(conf*100).toFixed(1)}%)`);
  }
};

recognition.onerror = (e) => console.error(e.error);
recognition.start();

📝 Dictation Mode

Continuous speech recognition. Say "period", "comma", "question mark", "exclamation mark", or "new line" to insert punctuation.

Press the mic button to begin dictating.

📄 Code Snippet

// Dictation mode with punctuation replacement
const recognition = new webkitSpeechRecognition();
recognition.continuous = true;
recognition.interimResults = true;
recognition.lang = 'en-US';

const punctuation = {
  'period': '.', 'full stop': '.',
  'comma': ',', 'question mark': '?',
  'exclamation mark': '!', 'exclamation point': '!',
  'new line': '\n', 'new paragraph': '\n\n',
};

recognition.onresult = (event) => {
  let text = event.results[event.resultIndex][0].transcript;
  if (event.results[event.resultIndex].isFinal) {
    for (const [word, char] of Object.entries(punctuation)) {
      text = text.replace(new RegExp(`\\b${word}\\b`, 'gi'), char);
    }
    document.getElementById('output').value += text;
  }
};

recognition.start();

// Round-trip: Recognize → Synthesize
async function roundTrip() {
  const recognition = new webkitSpeechRecognition();
  recognition.lang = 'en-US';
  recognition.interimResults = false;

  const text = await new Promise((resolve, reject) => {
    recognition.onresult = (e) => resolve(e.results[0][0].transcript);
    recognition.onerror = (e) => reject(e.error);
    recognition.start();
  });

  console.log('Recognized:', text);

  const utter = new SpeechSynthesisUtterance(text);
  utter.voice = speechSynthesis.getVoices()
    .find(v => v.lang === 'ja-JP'); // read back in Japanese
  speechSynthesis.speak(utter);
}

🌍 Language Comparison

Hear the same text spoken in multiple languages. Uses the best available voice per language.

Presets

📄 Code Snippet

// Speak in multiple languages sequentially
const languages = ['en-US', 'ja-JP', 'fr-FR', 'de-DE', 'es-ES'];

async function speakInLanguages(text) {
  const voices = speechSynthesis.getVoices();

  for (const lang of languages) {
    const voice = voices.find(v => v.lang === lang);
    if (!voice) continue;

    await new Promise(resolve => {
      const utter = new SpeechSynthesisUtterance(text);
      utter.voice = voice;
      utter.lang = lang;
      utter.onend = resolve;
      utter.onerror = resolve;
      speechSynthesis.speak(utter);
    });
  }
}

speakInLanguages('Hello, how are you?');
